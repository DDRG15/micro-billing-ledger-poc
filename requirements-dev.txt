# requirements-dev.txt — Development toolchain for posthog-billing-poc
# ======================================================================
# Install with: pip install -r requirements-dev.txt
#     These tools are NOT needed to run the app or the test suite.
#     They are for code quality checks during development.
#
#
# Usage / Uso:
#   ruff check ledger.py            # lint
#   ruff format ledger.py           # format
#   mypy ledger.py                  # type check
#   bandit -r ledger.py             # security scan

# ruff — extremely fast Python linter and formatter (replaces flake8 + isort + pyupgrade).
#     Detects unused imports, undefined names, style violations, and many more.
ruff==0.4.4

# mypy — static type checker. Enforces that type annotations (like Optional[int])
#     are actually respected in the code. Catches type mismatches before runtime.
mypy==1.10.0

# bandit — security-focused linter. Scans for hardcoded passwords, unsafe calls,
#     SQL injection patterns, and other OWASP Top 10 vulnerabilities in Python code.
bandit==1.7.8
