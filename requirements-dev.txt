# EN: ruff — extremely fast Python linter and formatter (replaces flake8 + isort + pyupgrade).
#     Detects unused imports, undefined names, style violations, and many more.
# ES: ruff — linter y formateador Python extremadamente rápido (reemplaza flake8 + isort + pyupgrade).
ruff==0.4.4

# EN: mypy — static type checker. Enforces that type annotations (like Optional[int])
#     are actually respected in the code. Catches type mismatches before runtime.
# ES: mypy — verificador de tipos estático. Aplica que las anotaciones de tipo
#     sean respetadas en el código. Detecta discrepancias de tipo antes de ejecución.

# EN: bandit — security-focused linter. Scans for hardcoded passwords, unsafe calls,
#     SQL injection patterns, and other OWASP Top 10 vulnerabilities in Python code.
# ES: bandit — linter enfocado en seguridad. Escanea contraseñas hardcodeadas,
#     llamadas inseguras, patrones de inyección SQL y otras vulnerabilidades OWASP Top 10.
bandit==1.7.8
