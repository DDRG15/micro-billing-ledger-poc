# EN: FastAPI 0.115.0 — HTTP framework for the webhook endpoint and summary/DLQ routes.
#     Chosen over Flask because it ships async support, automatic OpenAPI docs, and
#     Pydantic v2 integration out of the box. 0.115.0 is stable and tested.
# ES: FastAPI 0.115.0 — framework HTTP para el endpoint de webhook y rutas de resumen/DLQ.
#     Elegido sobre Flask porque incluye soporte async, docs OpenAPI automáticas, e
#     integración con Pydantic v2 fuera de la caja. 0.115.0 es estable y probado.
fastapi==0.115.0

# EN: uvicorn[standard] 0.30.6 — ASGI server that runs the FastAPI app.
#     [standard] includes uvloop (faster event loop, C extension) and httptools
#     (faster HTTP parsing). Both require the gcc/musl-dev in the builder Dockerfile stage.
#     For production with PostgreSQL and multiple workers, consider Gunicorn + uvicorn workers.
# ES: uvicorn[standard] 0.30.6 — servidor ASGI que ejecuta la app FastAPI.
#     [standard] incluye uvloop (event loop más rápido, extensión C) y httptools
#     (parseo HTTP más rápido). Ambos requieren gcc/musl-dev en la etapa builder del Dockerfile.
#     Para producción con PostgreSQL y múltiples workers, considerar Gunicorn + workers uvicorn.
uvicorn[standard]==0.30.6

# EN: Pydantic 2.9.2 — the entire validation stack runs on this.
#     v2 (not v1) is required because we use model_dump() (not deprecated dict()),
#     model_validator(mode='after'), and Field constraints with ge=0/pattern/min_length.
#     DO NOT downgrade to v1 — the API is incompatible and the validators will silently fail.
# ES: Pydantic 2.9.2 — toda la pila de validación corre sobre esto.
#     Se requiere v2 (no v1) porque usamos model_dump() (no el obsoleto dict()),
#     model_validator(mode='after'), y restricciones de Field con ge=0/pattern/min_length.
#     NO bajar de versión a v1 — la API es incompatible y los validadores fallarán silenciosamente.
pydantic==2.9.2
