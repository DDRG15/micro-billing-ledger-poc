# requirements.txt — Micro-Billing-Ledger PoC
# =============================================
# All production dependencies pinned to exact versions.
#     Pinning prevents "it worked yesterday" surprises when a new minor version
#     ships a breaking change. Update pins intentionally, not accidentally.
#     Install with: pip install -r requirements.txt
#

# FastAPI 0.115.0 — HTTP framework for the webhook endpoint and summary/DLQ routes.
#     Chosen over Flask because it ships async support, automatic OpenAPI docs, and
#     Pydantic v2 integration out of the box. 0.115.0 is stable and tested.
fastapi==0.115.0

# uvicorn[standard] 0.30.6 — ASGI server that runs the FastAPI app.
#     [standard] includes uvloop (faster event loop, C extension) and httptools
#     (faster HTTP parsing). Both require the gcc/musl-dev in the builder Dockerfile stage.
#     For production with PostgreSQL and multiple workers, consider Gunicorn + uvicorn workers.
uvicorn[standard]==0.30.6

# Pydantic 2.9.2 — the entire validation stack runs on this.
#     v2 (not v1) is required because we use model_dump() (not deprecated dict()),
#     model_validator(mode='after'), and Field constraints with ge=0/pattern/min_length.
#     DO NOT downgrade to v1 — the API is incompatible and the validators will silently fail.
pydantic==2.9.2

# Stripe 10.12.0 — SDK for webhook signature verification.
#     stripe.Webhook.construct_event() validates the HMAC-SHA256 signature in the
#     Stripe-Signature header. Without this, anyone who knows the URL can POST fake events.
#     Set STRIPE_WEBHOOK_SECRET in the environment to activate enforcement.
stripe==10.12.0

psycopg2-binary==2.9.9  # PostgreSQL sync driver — ACTIVE. Binary package includes libpq; no OS headers needed.

# slowapi 0.1.9 — rate limiting for FastAPI. Wraps the `limits` library.
#     Limits: 100 req/min on POST /webhook/stripe, 20 req/min on GET /dlq/entries
#     and GET /ledger/summary. Prevents connection pool exhaustion and DLQ flooding.
slowapi==0.1.9

# alembic 1.18.4 — database migration framework (Phase 10).
#     Replaces _bootstrap() CREATE TABLE IF NOT EXISTS for production schema management.
#     Migrations live in alembic/versions/. Run: alembic upgrade head
alembic==1.18.4

# SQLAlchemy 2.0.41 — required by alembic as its engine layer.
#     Only used for migration execution, not for ORM queries (psycopg2 handles all app queries).
sqlalchemy==2.0.41

# Uncomment when adding Prometheus /metrics endpoint.
